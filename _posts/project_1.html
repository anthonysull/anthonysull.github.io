# Function definition to calculate standard deviation
def stddev(lister):
    variance = [(i - np.mean(lister)) ** 2 for i in lister]
    variance = np.sum(variance)
    variance = variance / len(lister)
    std = variance**(1/2.0)
    return 'Std Dev', std, 'Var', variance
